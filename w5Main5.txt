

@startuml
title Conditional - Convert marks to grade 
start
:get user input A or B or C or D or F;
:get user input marks;
if (90<=marks<=100)
:grade=A;
elseif(80<=marks<90)
:grade=B;
elseif (70<=marks<80)
:grade=C;
elseif (60<=marks<70)
:grade=D;
elseif (0<=marks<60)
:grade=F;
endif
:print grade;
stop
@enduml
